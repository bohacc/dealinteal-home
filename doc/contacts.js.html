'use strict';

/**
 * @file Contacts
 * @fileOverview ContactsCtrl
 */

/**
 * @namespace ContactsCtrl
 * @author Martin Boháč
 */

angular.module('crmPostgresWebApp')
  .controller('ContactsCtrl', ['$scope', '$translate', '$translatePartialLoader', 'ContactsService', 'listContacts', function ($scope, $translate, $translatePartialLoader, ContactsService, listContacts) {
    $translatePartialLoader.addPart('contacts');
    $translate.refresh();

    $scope.searchStr = '';
    $scope.dataLoaderParams = {
      sortField: 'last_name',
      sortDirection: 'asc'
    };
    $scope.infoSummary = {};

    // list
    $scope.contacts = listContacts;
    $scope.dataLoader = ContactsService.list;
  }]);

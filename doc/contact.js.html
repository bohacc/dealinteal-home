'use strict';

/**
 * @file Contact
 * @fileOverview ContactCtrl
 */

/**
 * @namespace ContactCtrl
 * @author Martin Boháč
 */

angular.module('crmPostgresWebApp')
  .controller('ContactCtrl', ['$scope', '$translate', '$translatePartialLoader', 'contact', 'ContactFactory', '$location', '$routeParams', 'PageAncestor',
    function ($scope, $translate, $translatePartialLoader, contact, ContactFactory, $location, $routeParams, PageAncestor) {
      $translatePartialLoader.addPart('contact');
      $translate.refresh();

      $scope.contact = contact;

      /**
       * @memberof ContactCtrl
       * @method
       * @name putContact
       * @description put current Contact
       * @returns void
       */
      $scope.putContact = function () {
        PageAncestor.put(function () {
          return ContactFactory.put(contact);
        });
      };

      /**
       * @memberof ContactCtrl
       * @method
       * @name postContact
       * @description post current Contact
       * @returns void
       */
      $scope.postContact = function () {
        PageAncestor.post(function () {
          return ContactFactory.post(contact);
        });
      };

      /**
       * @memberof ContactCtrl
       * @method
       * @name saveContact
       * @description put or post current Contact
       * @returns void
       */
      $scope.saveContact = function () {
        if ($routeParams.id) {
          PageAncestor.put(function () {
            return ContactFactory.put($scope.contact);
          });
        } else {
          PageAncestor.post(function () {
            return ContactFactory.post($scope.contact);
          });
        }
      };

      /**
       * @memberof ContactCtrl
       * @method
       * @name deleteContact
       * @description delete current Contact
       * @returns void
       */
      $scope.deleteContact = function () {
        PageAncestor.del(function () {
          return ContactFactory.delete(contact).then(function () {
            $location.path('/contacts');
          });
        });
      };

      // Run
      PageAncestor.init({
        scope: $scope,
        formObject: 'contact',
        table: 'PEOPLE'
      });
    }]);
